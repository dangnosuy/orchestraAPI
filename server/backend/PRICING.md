# Công thức tính chi phí cho mỗi request:
#   cost = (prompt_tokens × input_price + completion_tokens × output_price) / 1,000
#   (input_price và output_price lưu trong DB theo đơn vị USD per 1K tokens)
#
# Ví dụ: User gọi Claude Sonnet 4.5, gửi 500 input tokens, nhận 1000 output tokens
#   input_price  = 0.002000 (per 1K)  →  $2.00 per 1M
#   output_price = 0.012000 (per 1K)  → $12.00 per 1M
#   cost = (500 × 0.002000 + 1000 × 0.012000) / 1000
#        = (1.0 + 12.0) / 1000
#        = $0.013
#
# Ví dụ: User gọi Claude Opus 4.6, gửi 2000 input tokens, nhận 3000 output tokens
#   cost = (2000 × 0.003333 + 3000 × 0.020000) / 1000
#        = (6.666 + 60.0) / 1000
#        = $0.0667
